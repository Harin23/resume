
--------- capitalize the first letter for multiple word searches--------------------------------------------

function detect_space_and_uppercase_fl(a){
    var user_i, user_ic, l;
    user_i=a;
    l=user_i.length;
    if (user_i[l-1] == " "){
        user_ic = user_i[l].toUpperCase();
        user_i[l]=user_ic;
    }
    return user_i;
}
